example TC read channel data with logger

  1'''
  2Temperature-TC - example_TC_read_channel_data_with_logger.py
  3
  4This example demonstrates how to read thermocouple and save data into csv file from WPC-USB-DAQ-F1-TD.
  5
  6First, it shows how to open thermal port and configure thermal parameters.
  7Second, read channel 1 thermocouple data and save them. 
  8Last, close thermal port.
  9
 10For other examples please check:
 11   https://github.com/WPC-Systems-Ltd/WPC_Python_driver_release/tree/main/Examples
 12
 13   See README.md file to get detailed usage of this example.
 14
 15Copyright (c) 2022 WPC Systems Ltd.
 16All rights reserved.
 17
 18'''
 19
 20## Python
 21import asyncio
 22
 23## WPC
 24from wpcsys import pywpc
 25
 26async def main():
 27    print("Start example code...")
 28
 29    ## Get Python driver version
 30    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
 31
 32    ## Create datalogger handle 
 33    dev_logger = pywpc.DataLogger()
 34
 35    ## Open file with WPC_test.csv
 36    dev_logger.Logger_openFile("WPC_test.csv")
 37
 38    ## Write header into CSV file
 39    dev_logger.Logger_writeHeader(["Thermo CH1"])
 40
 41    ## Create device handle
 42    dev = pywpc.USBDAQF1TD()
 43
 44    ## Connect to USB device
 45    try:
 46        dev.connect('21JA1239')
 47    except Exception as err:
 48        pywpc.printGenericError(err)
 49
 50    try: 
 51        ## Get firmware model & version
 52        driver_info = await dev.Sys_getDriverInfo_async()
 53        print("Model name: " + driver_info[0])
 54        print("Firmware version: " + driver_info[-1])
 55        
 56        ## Parameters setting
 57        port = 1
 58        channel = 1 
 59        over_sampling_mode = 0  ## 0:1 sample, 1:2 samples, 2:4 sample, 3:8 samples, 4:16 samples
 60        thermo_type = 3         ## 0:B type, 1:E type, 2:J type, 3:K type
 61                                ## 4:N type, 5:R type, 6:S type, 7:T type
 62        
 63        ## Open thermo port1
 64        status = await dev.Thermal_open_async(port)
 65        if status == 0: print("Thermal_open: OK")
 66
 67        ## Sleep
 68        await asyncio.sleep(0.1) ## delay(second)
 69        
 70        ## Set thermo port to 1 and set over-sampling mode to no over-sampling in channel 1 
 71        status = await dev.Thermal_setOverSampling_async(port, channel, over_sampling_mode)
 72        if status == 0: print("setOverSampling: OK")   
 73
 74        ## Sleep
 75        await asyncio.sleep(0.1) ## delay(second)
 76        
 77        ## Set thermo port to 1 and set K type in channel 1 
 78        status = await dev.Thermal_setType_async(port, channel, thermo_type)
 79        if status == 0: print("setType: OK")   
 80 
 81        ## Sleep
 82        await asyncio.sleep(0.1) ## delay(second)
 83
 84        ## Set thermo port to 1 and read thermo in channel 1
 85        data = await dev.Thermal_readSensor_async(port, channel)
 86        print("Read channel 1 data:", data, "°C")
 87
 88        ## Write data into CSV file
 89        dev_logger.Logger_writeValue(data)
 90
 91        ## Close thermo port1
 92        status = await dev.Thermal_close_async(port)
 93        if status == 0: print("Thermal_close: OK")
 94
 95        ## Close File
 96        dev_logger.Logger_closeFile()  
 97    except Exception as err:
 98        pywpc.printGenericError(err)
 99
100    ## Disconnect USB device
101    dev.disconnect()
102
103    ## Release device handle
104    dev.close()
105
106    print("End example code...")
107    return
108    
109if __name__ == '__main__':
110    asyncio.run(main())
