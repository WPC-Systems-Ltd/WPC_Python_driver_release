 1import asyncio
 2import sys
 3sys.path.insert(0, 'pywpc/')
 4sys.path.insert(0, '../../../pywpc/')
 5import pywpc
 6
 7async def main():
 8    print("Start example code...")
 9
10    ## Get Python driver version
11    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
12
13    ## Create device handle
14    dev = pywpc.USBDAQF1D()
15
16    ## Connect to network device
17    try:
18        dev.connect('21JA1044')
19    except Exception as err:
20        pywpc.printGenericError(err)
21
22    try: 
23        ## Get firmware model & version
24        driver_info = await dev.sys_getDriverInfo()
25        print("Model name: " + driver_info[0])
26        print("Firmware version: " + driver_info[-1])
27
28        port = 0
29        ## Open pin0, pin1, pin2, pin3 and pin4 in port 0 with digital output
30        await dev.DO_openPins(port, [0,1,2,3,4]) 
31
32        ## Set pin0, pin1 to high, others to low.
33        await dev.DO_writeValuePins(port, [0,1,2,3,4], [1,0,0,0,0]) 
34        
35        ## Open pin5, pin6 and pin7 in port 0 with digital output
36        await dev.DI_openPins(port, [5,6,7])
37
38        ## Read pin5, pin6 and pin7 state in port 0
39        state_list = await dev.DI_readPins(port, [7,5,6])
40        print(state_list)
41
42        ## Wait for 3 seconds
43        await asyncio.sleep(3) ## delay(second)
44
45    except Exception as err:
46        pywpc.printGenericError(err)
47
48    ## Close pin0, pin1, pin2, pin3 and pin4 in port 0 with digital output 
49    await dev.DO_closePins(port, [0,1,2,3,4])
50
51    ## Close pin5, pin6 and pin7 in port 0 with digital input
52    await dev.DI_closePins(port, [5,6,7])
53 
54    ## Disconnect network device
55    dev.disconnect()
56    
57    ## Release device handle
58    dev.close()
59
60    print("End example code...")
61    return
62
63if __name__ == '__main__':
64    asyncio.run(main())
