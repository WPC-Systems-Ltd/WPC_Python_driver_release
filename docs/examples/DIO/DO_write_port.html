 1'''
 2DIO - example_DO_write_port.py
 3
 4This example demonstrates how to write DO in port from WPC-USB-DAQ-F1-D.
 5
 6First, it shows how to open DO in port.
 7Second, write DO pins in two different types (hex or list) but it should be consistency.
 8Last, close DO in port.
 9
10For other examples please check:
11   https://github.com/WPC-Systems-Ltd/WPC_Python_driver_release/tree/main/Examples
12
13   See README.md file to get detailed usage of this example.
14
15Copyright (c) 2022 WPC Systems Ltd.
16All rights reserved.
17
18'''
19
20## Python
21import asyncio
22
23## WPC
24from wpcsys import pywpc
25
26async def main():
27    print("Start example code...")
28
29    ## Get Python driver version
30    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
31
32    ## Create device handle
33    dev = pywpc.USBDAQF1D()
34
35    ## Connect to USB device
36    try:
37        dev.connect('21JA1044')
38    except Exception as err:
39        pywpc.printGenericError(err)
40
41    try: 
42        ## Get firmware model & version
43        driver_info = await dev.Sys_getDriverInfo_async()
44        print("Model name: " + driver_info[0])
45        print("Firmware version: " + driver_info[-1])
46        
47        ## Parameters setting
48        port = 0
49        
50        ## Open all pins in port 0 and set it to digital output
51        status = await dev.DO_openPort_async(port)
52        if status == 0: print("DO_openPort: OK")
53        
54        ## Set pin0, pin3 and pin4 to high, others to low
55        status = await dev.DO_writePort_async(port, [0,0,0,1,1,0,0,1])
56        if status == 0: print("DO_writePort: OK")
57
58        ## Wait for 5 second
59        await asyncio.sleep(5)  ## delay(second)
60
61        ## Set pin7 and pin6 to high, others to low (1100 0000 in binary) (0xC0 in hex).
62        status = await dev.DO_writePort_async(port, 0xC0)
63        if status == 0: print("DO_writePort: OK")
64       
65        ## Wait for 5 second
66        await asyncio.sleep(5)  ## delay(second)
67
68        ## Close all pins in port 0 with digital output
69        status = await dev.DO_closePort_async(port)
70        if status == 0: print("DO_closePort: OK")
71    except Exception as err:
72        pywpc.printGenericError(err)
73        
74    ## Disconnect USB device
75    dev.disconnect()
76    
77    ## Release device handle
78    dev.close()
79
80    print("End example code...")
81    return
82
83if __name__ == '__main__':
84    asyncio.run(main())
