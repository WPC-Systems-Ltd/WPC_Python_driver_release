  1import asyncio
  2import sys
  3sys.path.insert(0, 'pywpc/')
  4sys.path.insert(0, '../../../pywpc/')
  5import pywpc
  6
  7async def main():
  8    print("Start example code...")
  9
 10    ## Get Python driver version
 11    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
 12
 13    ## Create device handle
 14    dev = pywpc.USBDAQF1D()
 15
 16    ## Connect to network device
 17    try:
 18        dev.connect("21JA0514")
 19    except Exception as err:
 20        pywpc.printGenericError(err)
 21        
 22    try: 
 23        ## Get firmware model & version
 24        driver_info = await dev.Sys_getDriverInfo()
 25        print("Model name: " + driver_info[0])
 26        print("Firmware version: " + driver_info[-1])
 27        
 28        ## Parameters setting
 29        SPI_port = 1
 30        datasize = 0  ## Mode: 0 = 8-bit data, 1 = 16-bit data.
 31        first_bit = 0 ## Mode: 0 = MSB first, 1 = LSB first.
 32        prescaler = 4
 33        cpol = 0 
 34        cpha = 0
 35
 36        ## Open DO pins. For cs, convert and reset pin
 37        DO_port_1 = 1 
 38        DO_index_1 = [4,5] ## pin4 for convert, pin5 for reset  
 39        DO_port_2 = 2 
 40        DO_index_2 = [0] ## CS pin 
 41
 42        ## Open pin4 and pin5 in port 1 with digital output
 43        status = await dev.DO_openPins(DO_port_1, DO_index_1) 
 44        if status == 0: print("DO_openPins: OK")
 45 
 46        ## Open pin0 in port 2 with digital output
 47        status = await dev.DO_openPins(DO_port_2, DO_index_2) 
 48        if status == 0: print("DO_openPins: OK")
 49
 50        ## Set pin4 and pin5 to high, others to low.
 51        status = await dev.DO_writeValuePins(DO_port_1, DO_index_1, [1, 1])
 52        if status == 0: print("DO_writeValuePins: OK")
 53
 54        ## Set pin0 to high, others to low.
 55        status = await dev.DO_writeValuePins(DO_port_2, DO_index_2, [1])
 56        if status == 0: print("DO_writeValuePins: OK")
 57
 58        ## Open SPI port1
 59        status = await dev.SPI_open(SPI_port)
 60        if status == 0: print("SPI_open: OK")
 61        
 62        ## Set SPI port to 1 and set datasize to 8-bit data
 63        status = await dev.SPI_setDataSize(SPI_port, datasize)
 64        if status == 0: print("SPI_setDataSize: OK")
 65        
 66        ## Set SPI port to 1 and set first_bit to MSB first
 67        status = await dev.SPI_setFirstBit(SPI_port, first_bit)
 68        if status == 0: print("SPI_setFirstBit: OK")
 69        
 70        ## Set SPI port to 1 and set prescaler to 4
 71        status = await dev.SPI_setPrescaler(SPI_port, prescaler)
 72        if status == 0: print("SPI_setPrescaler: OK")
 73       
 74        ## Set SPI port to 1 and set CPOL to 0
 75        status = await dev.SPI_setCPOL(SPI_port, cpol)
 76        if status == 0: print("SPI_setCPOL: OK")
 77        
 78        ## Set SPI port to 1 and set CPHA to 0
 79        status = await dev.SPI_setCPHA(SPI_port, cpha)
 80        if status == 0: print("SPI_setCPHA: OK")
 81
 82        ## Toggle reset pin 
 83        await dev.DO_writeValuePins(DO_port_1, 5, 1) ## pin5 for reset 
 84        await dev.DO_writeValuePins(DO_port_1, 5, 0) 
 85        
 86        ## Toggle convert pin 
 87        await dev.DO_writeValuePins(DO_port_1, 4, 0) ## pin4 for convert 
 88        await dev.DO_writeValuePins(DO_port_1, 4, 1)
 89         
 90        ## Set cs pin (pin0) to low
 91        status = await dev.DO_writeValuePins(DO_port_2, DO_index_2, [0]) 
 92        if status == 0: print("DO_writeValuePins: OK")
 93       
 94        data = await dev.SPI_read(SPI_port, 16)
 95        data = ['{:02x}'.format(value) for value in data]
 96        print("data :", data)
 97     
 98        ## Set cs pin (pin0) to high
 99        status = await dev.DO_writeValuePins(DO_port_2, DO_index_2, [1])
100        if status == 0: print("DO_writeValuePins: OK")
101         
102        ## Close pin4 and pin5 in port 1 with digital output
103        status = await dev.DO_closePins(DO_port_1, DO_index_1) 
104        if status == 0: print("DO_closePins: OK")
105 
106        ## Close pin0 in port 2 with digital output
107        status = await dev.DO_closePins(DO_port_2, DO_index_2) 
108        if status == 0: print("DO_closePins: OK")
109
110        ## Close SPI port1
111        status = await dev.SPI_close(SPI_port)       
112        if status == 0: print("SPI_close: OK")
113    except Exception as err:
114        pywpc.printGenericError(err)
115
116    ## Disconnect network device
117    dev.disconnect()
118
119    ## Release device handle
120    dev.close()
121
122    print("End example code...")
123    return
124    
125if __name__ == '__main__':
126    asyncio.run(main())
