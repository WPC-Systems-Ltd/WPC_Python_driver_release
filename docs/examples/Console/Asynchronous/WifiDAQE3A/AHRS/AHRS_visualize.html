 49def WPC_initializeFigure(nb_rows=1, nb_col=1, share_x='all', share_y='all', option='none'):
 50  fig = plt.gcf()
 51  fig.clf()
 52
 53  ## If `option` is `3D`, return a 3-D axis.
 54  if option == '3D':
 55    ax = fig.add_subplot(1, 1, 1, projection='3d')
 56    return fig, ax, None, None
 57
 58  ## If `option` is `grid`, return gridspec for further usage.
 59  if option == 'grid':
 60    ax_grid = mgs.GridSpec(nb_rows, nb_col, figure=fig)
 61    return fig, None, None, ax_grid
 62
 63  ## If empty rows or columns, return `None`.
 64  if nb_rows == 0 or nb_col == 0:
 65    return fig, None, None, None
 66
 67  ## Return subplots
 68  ax_mat = fig.subplots(nb_rows, nb_col, sharex=share_x, sharey=share_y, squeeze=False)
 69  ax_arr = ax_mat.flatten()
 70  ax = ax_arr[0]
 71  return fig, ax, ax_arr, ax_mat

 73def WPC_saveFigure(save, fig, tag, prefix='', verbose=True):
 74  if save < 0:
 75    tag = prefix + tag
 76
 77  save = abs(save)
 78
 79  if save == 2:
 80    fig.patch.set_alpha(0.5)
 81    name = f'{tag}.pdf'
 82    fig.savefig(name)
 83
 84    if verbose:
 85      print(f'Saved \"{name}\"')
 86
 87  if save in [1, 2]:
 88    fig.patch.set_alpha(1.0)
 89    name = f'{tag}.png'
 90    fig.savefig(name)
 91
 92    if verbose:
 93      print(f'Saved \"{name}\"')
 94
 95  if save == 0:
 96    w, h = fig.get_size_inches()
 97    fig.set_size_inches(w, h, forward=True)
 98    mpl.pyplot.ion()
 99    mpl.pyplot.show()
100  return

102def WPC_getRotMat(roll, pitch, yaw, use_deg=False):
103  if use_deg:
104    roll *= DEGREE_TO_RADIAN
105    pitch *= DEGREE_TO_RADIAN
106    yaw *= DEGREE_TO_RADIAN
107  rot_mat_x = np.array([[1, 0, 0], [0, np.cos(roll), np.sin(roll)], [0, -np.sin(roll), np.cos(roll)]])
108  rot_mat_y = np.array([[np.cos(pitch), 0, np.sin(pitch)], [0, 1, 0], [-np.sin(pitch), 0, np.cos(pitch)]])
109  rot_mat_z = np.array([[np.cos(yaw), -np.sin(yaw), 0], [np.sin(yaw), np.cos(yaw), 0], [0, 0, 1]])
110  rot_mat = rot_mat_x.dot(rot_mat_y).dot(rot_mat_z) ## Dot product from the back
111  return rot_mat

113def WPC_showEmpty(tag=DEFAULT_MODEL, save=0):
114  fig, ax, _, _ = WPC_initializeFigure(option='3D')
115
116  d = MODEL_DICT[tag]
117
118  ## Load
119  model = mesh.Mesh.from_file(f'{DATA_PATH}{tag}.stl')
120  data_orig = model.vectors
121
122  ## Plot
123  poly = mplot3d.art3d.Poly3DCollection([], color='orange', edgecolor='k', lw=0.2)
124  ax.add_collection3d(poly)
125
126  ## Settings
127  scale = [-0.6*d['view'], 0.6*d['view']]
128  ax.view_init(elev=0, azim=0, roll=0)
129  ax.auto_scale_xyz(scale, scale, scale)
130  ax.set_axis_off()
131  ax.set_title(f'sensor fusion {tag}')
132  ax.text2D(0.7, 0.95, "Roll:", transform=ax.transAxes)
133  ax.text2D(0.7, 0.9, "Pitch:", transform=ax.transAxes)
134  ax.text2D(0.7, 0.85, "Yaw:", transform=ax.transAxes)
135
136  ## Save
137  fig.set_size_inches(6, 6)
138  fig.subplots_adjust(left=0.05, right=0.95, bottom=0.05, top=0.95)
139  fig.canvas.manager.set_window_title('WPC AHRS visualization')
140
141  WPC_saveFigure(save, fig, f'{IMG_PATH}{tag}_empty')
142  return fig, ax, data_orig, poly

144def WPC_drawCat(fig, ax, data_orig, poly, roll, pitch, yaw):
145  ## Rotate
146  rot_mat = WPC_getRotMat(roll, pitch, yaw, use_deg=True)
147  data = data_orig.dot(rot_mat)
148
149  Roll = "{:7.2f}".format(roll)
150  Pitch = "{:7.2f}".format(pitch)
151  Yaw = "{:7.2f}".format(yaw)
152
153  tx1 = ax.text2D(0.78, 0.95, Roll, transform=ax.transAxes)
154  tx2 = ax.text2D(0.78, 0.9, Pitch, transform=ax.transAxes)
155  tx3 = ax.text2D(0.78, 0.85, Yaw, transform=ax.transAxes)
156
157  ## Plot
158  poly.set_verts(data)
159  fig.canvas.flush_events()
160
161  tx1.remove()
162  tx2.remove()
163  tx3.remove()
164  return

166async def main():
167    ## Get Python driver version
168    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
169
170    ## Create device handle
171    dev = pywpc.WifiDAQE3A()
172
173    ## Show empty
174    fig, ax, data_orig, poly = WPC_showEmpty()
175
176    ## Connect to device
177    try:
178        dev.connect("192.168.5.38") ## Depend on your device
179    except Exception as err:
180        pywpc.printGenericError(err)
181        ## Release device handle
182        dev.close()
183        return
184
185    try:
186        ## Parameters setting
187        port = 0 ## Depend on your device
188        sampling_period = 0.003
189        read_delay = 0.5 ## second
190
191        ## Get firmware model & version
192        driver_info = await dev.Sys_getDriverInfo_async()
193        print("Model name: " + driver_info[0])
194        print("Firmware version: " + driver_info[-1])
195
196        ## Open port
197        err = await dev.AHRS_open_async(port)
198        print(f"AHRS_open_async in port {port}: {err}")
199
200        ## Set period
201        err = await dev.AHRS_setSamplingPeriod_async(port, sampling_period)
202        print(f"AHRS_setSamplingPeriod_async in port {port}: {err}")
203
204        ## Start AHRS
205        err = await dev.AHRS_start_async(port)
206        print(f"AHRS_start_async in port {port}: {err}")
207
208        while True:
209            ahrs_list = await dev.AHRS_readStreaming_async(port, read_delay)
210            if len(ahrs_list) > 0:
211                WPC_drawCat(fig, ax, data_orig, poly, ahrs_list[0], ahrs_list[1], ahrs_list[2])
212
213    except KeyboardInterrupt:
214        print("Press keyboard")
215
216    except Exception as err:
217        pywpc.printGenericError(err)
218
219    finally:
220        ## Stop AHRS
221        err = await dev.AHRS_stop_async(port)
222        print(f"AHRS_stop_async in port {port}: {err}")
223
224        ## Close AHRS
225        err = await dev.AHRS_close_async(port)
226        print(f"AHRS_close_async in port {port}: {err}")
227
228        ## Disconnect device
229        dev.disconnect()
230
231        ## Release device handle
232        dev.close()
233    return
