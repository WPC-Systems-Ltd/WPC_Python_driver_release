 1'''
 2Tutorial - single_loop_async.py
 3
 4This example project demonstrates how to use async thread to get RTC from USBDAQF1D.
 5
 6For other examples please check:
 7    https://github.com/WPC-Systems-Ltd/WPC_Python_driver_release/tree/main/examples
 8See README.md file to get detailed usage of this example.
 9
10Copyright (c) 2022 WPC Systems Ltd.
11All rights reserved.
12'''
13
14## Python
15
16import asyncio
17
18## WPC
19
20from wpcsys import pywpc
21
22async def loop_func(handle, timeout = 3, delay = 1):
23    t = 0
24    while t < timeout: ## timeout(second)
25        data = await handle.Sys_getRTC_async()
26        print("RTC Time: " + str(data))
27        await asyncio.sleep(delay)  ## delay(second)
28        t += delay
29
30async def main():
31    print("Start example code...")
32
33    ## Get Python driver version
34    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
35
36    ## Create device handle
37    dev = pywpc.USBDAQF1D()
38
39    ## Connect to device
40    try:
41        dev.connect("21JA1200")
42    except Exception as err:
43        pywpc.printGenericError(err)
44    ## Perform async thread to query data
45    try:
46        ## Get firmware model & version
47        driver_info = await dev.Sys_getDriverInfo_async()
48        print("Model name: " + driver_info[0])
49        print("Firmware version: " + driver_info[-1])
50
51        await loop_func(dev, 3, 1) ## timeout, delay(second)
52    except Exception as err:
53        pywpc.printGenericError(err)
54
55    ## This part never execute because the async thread
56    ## Disconnect device
57    dev.disconnect()
58
59    ## Release device handle
60    dev.close()
61
62    print("End example code...")
63    return
64
65if __name__ == '__main__':
66    asyncio.run(main())
