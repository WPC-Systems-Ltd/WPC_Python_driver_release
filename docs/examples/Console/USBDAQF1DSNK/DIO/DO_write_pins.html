 1'''
 2DIO - DO_write_pins.py
 3
 4This example demonstrates how to write DO in pins from USBDAQF1DSNK.
 5
 6First, it shows how to open DO in pins.
 7Second, write DO pin in two different types (hex or list) but it should be consistency.
 8Last, close DO in pins.
 9
10For other examples please check:
11    https://github.com/WPC-Systems-Ltd/WPC_Python_driver_release/tree/main/examples
12See README.md file to get detailed usage of this example.
13
14Copyright (c) 2022 WPC Systems Ltd.
15All rights reserved.
16'''
17
18## Python
19
20import asyncio
21
22## WPC
23
24from wpcsys import pywpc
25
26async def main():
27    print("Start example code...")
28
29    ## Get Python driver version
30    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
31
32    ## Create device handle
33    dev = pywpc.USBDAQF1DSNK()
34
35    ## Connect to device
36    try:
37        dev.connect("21JA1298")
38    except Exception as err:
39        pywpc.printGenericError(err)
40
41    try:
42        ## Get firmware model & version
43        driver_info = await dev.Sys_getDriverInfo_async()
44        print("Model name: " + driver_info[0])
45        print("Firmware version: " + driver_info[-1])
46
47        ## Parameters setting
48        port = 0
49        pin_index = [0,1,2,3,4]
50
51        ## Open pin0, pin1, pin2, pin3 and pin4 with digital output.
52        status = await dev.DO_openPins_async(port, pin_index)
53        print("DO_openPins_async status: ", status)
54
55        ## Wait for 1 second
56        await asyncio.sleep(1)  ## delay(second)
57
58        ## Set pin0, pin1 to high, others to low.
59        status = await dev.DO_writePins_async(port, pin_index, [1,1,0,0,0])
60        print("DO_writePins_async status: ", status)
61
62        ## Open pin5, pin6 and pin7 with digital output (1110 0000 in binary) (0xE0 in hex).
63        status = await dev.DO_openPins_async(port, 0xE0)
64        print("DO_openPins_async status: ", status)
65
66        ## Wait for 1 second
67        await asyncio.sleep(1)  ## delay(second)
68
69        ## Set pin7 and pin6 to high, others to low (1100 0000 in binary) (0xC0 in hex).
70        status = await dev.DO_writePins_async(port, 0xE0, 0xC0)
71        print("DO_writePins_async status: ", status)
72
73        ## Wait for 5 second
74        await asyncio.sleep(5)  ## delay(second)
75
76        ## Close pin0, pin1, pin2, pin3 and pin4 with digital output.
77        status = await dev.DO_closePins_async(port, pin_index)
78        print("DO_closePins_async status: ", status)
79
80        ## Close pin5, pin6 and pin7 with digital output.
81        status = await dev.DO_closePins_async(port, 0xE0)
82        print("DO_closePins_async status: ", status)
83    except Exception as err:
84        pywpc.printGenericError(err)
85
86    ## Disconnect device
87    dev.disconnect()
88
89    ## Release device handle
90    dev.close()
91
92    print("End example code...")
93    return
94
95if __name__ == '__main__':
96    asyncio.run(main())
