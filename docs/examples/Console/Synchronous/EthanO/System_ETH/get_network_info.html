 1'''
 2System_ETH - get_network_info.py with synchronous mode.
 3
 4This example demonstrates how to get hardware & network information from EthanO.
 5
 6First, get hardware information such as firmware model & version.
 7Last, get network information such as IP & submask & MAC.
 8
 9-------------------------------------------------------------------------------------
10Please change correct serial number or IP and port number BEFORE you run example code.
11
12For other examples please check:
13    https://github.com/WPC-Systems-Ltd/WPC_Python_driver_release/tree/main/examples
14See README.md file to get detailed usage of this example.
15
16Copyright (c) 2023 WPC Systems Ltd. All rights reserved.
17'''
18
19## Python
20import time
21
22## WPC
23
24from wpcsys import pywpc
25
26def main():
27    ## Get Python driver version
28    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
29
30    ## Create device handle
31    dev = pywpc.EthanO()
32
33    ## Connect to device
34    try:
35        dev.connect("192.168.1.110") ## Depend on your device
36    except Exception as err:
37        pywpc.printGenericError(err)
38        ## Release device handle
39        dev.close()
40        return
41
42    ## Perform DAQ basic information
43    try:
44        ## Parameters setting
45        port = None ## Depend on your device
46        timeout = 3  ## second
47
48        ## Get firmware model & version
49        driver_info = dev.Sys_getDriverInfo(timeout=timeout)
50        print("Model name: " + driver_info[0])
51        print("Firmware version: " + driver_info[-1])
52
53        ## Get IP & submask
54        ip_addr, submask = dev.Net_getIPAddrAndSubmask(timeout=timeout)
55        print(f"IP: {ip_addr}")
56        print(f"Submask: {submask}")
57
58        ## Get MAC
59        mac = dev.Net_getMACAddr(timeout=timeout)
60        print(f"MAC: {mac}")
61    except Exception as err:
62        pywpc.printGenericError(err)
63
64    ## Disconnect network device
65    dev.disconnect()
66
67    ## Release device handle
68    dev.close()
69
70    return
71
72if __name__ == '__main__':
73    main()
