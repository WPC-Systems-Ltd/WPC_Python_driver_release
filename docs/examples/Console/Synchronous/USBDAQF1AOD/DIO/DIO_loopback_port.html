 1'''
 2DIO - DIO_loopback_port.py with synchronous mode.
 3
 4This example demonstrates how to write DIO loopback in port from USBDAQF1AOD.
 5Use DO pins to send signals and use DI pins to receive signals on single device also called "loopback".
 6
 7First, it shows how to open DO and DI in port.
 8Second, write DO in port and read DI in port
 9Last, close DO and DI in port.
10
11Please change correct serial number or IP and port number BEFORE you run example code.
12
13For other examples please check:
14    https://github.com/WPC-Systems-Ltd/WPC_Python_driver_release/tree/main/examples
15See README.md file to get detailed usage of this example.
16
17Copyright (c) 2023 WPC Systems Ltd. All rights reserved.
18'''
19
20## Python
21
22import time
23
24## WPC
25
26from wpcsys import pywpc
27
28def main():
29    ## Get Python driver version
30    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
31
32    ## Create device handle
33    dev = pywpc.USBDAQF1AOD()
34
35    ## Connect to device
36    try:
37        dev.connect("default") ## Depend on your device
38    except Exception as err:
39        pywpc.printGenericError(err)
40        ## Release device handle
41        dev.close()
42        return
43
44    try:
45        ## Parameters setting
46        port = 0 ## Depend on your device
47        port_DO = 0
48        port_DI = 1
49        timeout = 3  ## second
50
51        ## Get firmware model & version
52        driver_info = dev.Sys_getDriverInfo(timeout=timeout)
53        print("Model name: " + driver_info[0])
54        print("Firmware version: " + driver_info[-1])
55
56        ## Open all pins with digital output
57        err = dev.DO_openPort(port_DO, timeout=timeout)
58        print(f"DO_openPort in port {port_DO}: {err}")
59
60        ## Open all pins with digital input
61        err = dev.DI_openPort(port_DI, timeout=timeout)
62        print(f"DI_openPort in port {port_DI}: {err}")
63
64        ## Set pin0, pin1 and pin2 to high, others to low
65        err = dev.DO_writePort(port_DO, [0,0,0,1,0,0,0,0], timeout=timeout)
66        print(f"DO_writePort in port {port_DO}: {err}")
67
68        ## Read all pins state
69        state_list = dev.DI_readPort(port_DI, timeout=timeout)
70        print(f"state_list in port {port_DI}: {state_list}")
71
72        ## Close all pins with digital output
73        err = dev.DO_closePort(port_DO, timeout=timeout)
74        print(f"DO_closePort in port {port_DO}: {err}")
75
76        ## Close all pins with digital input
77        err = dev.DI_closePort(port_DI, timeout=timeout)
78        print(f"DI_closePort in port {port_DI}: {err}")
79    except Exception as err:
80        pywpc.printGenericError(err)
81
82    ## Disconnect device
83    dev.disconnect()
84
85    ## Release device handle
86    dev.close()
87
88    return
89
90if __name__ == '__main__':
91    main()
