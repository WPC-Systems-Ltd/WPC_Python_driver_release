  1##  Example_get_device_info/ main.py
  2##  This is an example for getting information with WPC DAQ Device with asynchronous mode.
  3##  Copyright (c) 2023 WPC Systems Ltd.
  4##  All rights reserved.
  5
  6## Python
  7import asyncio
  8import os
  9from qasync import QEventLoop, asyncSlot
 10
 11## Third party
 12from PyQt5 import QtWidgets, QtGui
 13from PyQt5.QtWidgets import QMessageBox
 14from UI_design.Ui_example_GUI_get_device_info import Ui_MainWindow
 15
 16## WPC
 17from wpcsys import pywpc
 18
 19class MainWindow(QtWidgets.QMainWindow):
 20    def __init__(self):
 21        super(MainWindow, self).__init__()
 22
 23        ## UI initialize
 24        self.ui = Ui_MainWindow()
 25        self.ui.setupUi(self)
 26
 27        ## Get Python driver version
 28        print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
 29
 30        ## Connection flag
 31        self.connect_flag = 0
 32
 33        ## Handle declaration
 34        self.dev = None
 35
 36        ## Material path
 37        file_path = os.path.dirname(__file__)
 38        self.trademark_path = file_path + "\Material\\trademark.jpg" 
 39        self.blue_led_path = file_path + "\Material\LED_blue.png"
 40        self.red_led_path = file_path + "\Material\LED_red.png"
 41        self.green_led_path = file_path + "\Material\LED_green.png"
 42        self.gray_led_path = file_path + "\Material\LED_gray.png"
 43
 44        ## Set trademark & LED path
 45        self.ui.lb_trademark.setPixmap(QtGui.QPixmap(self.trademark_path))
 46        self.ui.lb_led.setPixmap(QtGui.QPixmap(self.gray_led_path))
 47
 48        ## Define callback events
 49        self.ui.btn_connect.clicked.connect(self.connectEvent)
 50        self.ui.btn_disconnect.clicked.connect(self.disconnectEvent)
 51        self.ui.btn_deviceInfo.clicked.connect(self.getdeviceinfoEvent)
 52
 53    def selectHandle(self):
 54        handle_idx = int(self.ui.comboBox_handle.currentIndex())
 55        if handle_idx == 0:
 56            self.dev = pywpc.WifiDAQE3A()
 57        elif handle_idx == 1:
 58            self.dev = pywpc.EthanA()
 59        elif handle_idx == 2:
 60            self.dev = pywpc.EthanD()
 61
 62    def updateParam(self):
 63        ## Get IP or serial_number from GUI
 64        self.ip = self.ui.lineEdit_IP.text()
 65
 66    @asyncSlot()
 67    async def connectEvent(self):
 68        if self.connect_flag == 1:
 69            return
 70
 71        ## Select handle
 72        self.selectHandle()
 73
 74        ## Update Param
 75        self.updateParam()
 76
 77        ## Connect to device
 78        try:
 79            self.dev.connect(self.ip)
 80        except pywpc.Error as err:
 81            print("err: " + str(err))
 82            return
 83
 84        ## Change LED status
 85        self.ui.lb_led.setPixmap(QtGui.QPixmap(self.blue_led_path))
 86
 87        ## Change connection flag
 88        self.connect_flag = 1
 89
 90    @asyncSlot()
 91    async def disconnectEvent(self):
 92        if self.connect_flag == 0:
 93            return
 94
 95        ## Disconnect device
 96        self.dev.disconnect()
 97
 98        ## Change LED status
 99        self.ui.lb_led.setPixmap(QtGui.QPixmap(self.green_led_path))
100
101        ## Change connection flag
102        self.connect_flag = 0
103
104    def closeEvent(self, event):
105        if self.dev is not None:
106            ## Disconnect device
107            self.dev.disconnect()
108
109            ## Release device handle
110            self.dev.close()
111
112    @asyncSlot()
113    async def getdeviceinfoEvent(self):
114        ## Check connection status
115        if self.checkConnectionStatus() == False:
116            return
117
118        ## Get firmware model & version
119        driver_info = await self.dev.Sys_getDriverInfo_async()
120        model = driver_info[0]
121        version = driver_info[-1]
122
123        ## Get serial number & RTC Time
124        serial_number = await self.dev.Sys_getSerialNumber_async()
125        rtc = await self.dev.Sys_getRTC_async()
126
127        ## Get IP & submask & MAC
128        ip, submask = await self.dev.Sys_getIPAddrAndSubmask_async() 
129        mac = await self.dev.Sys_getMACAddr_async()
130
131        ## Update information in GUI
132        self.ui.lineEdit_ip.setText(ip)
133        self.ui.lineEdit_sbk.setText(submask)
134        self.ui.lineEdit_serialNum.setText(serial_number)
135        self.ui.lineEdit_mac.setText(mac)
136        self.ui.lineEdit_model.setText(model)
137        self.ui.lineEdit_version.setText(version)
138        self.ui.lineEdit_rtc.setText(rtc)
139
140    ## Check TCP connection with QMessageBox
141    def checkConnectionStatus(self):
142        if self.connect_flag == 0:
143            QMessageBox.information(self, "Error Messages", "Please connect server first.", QMessageBox.Ok)
144            return False
145        else:
146            return True
147
148def main():
149    app = QtWidgets.QApplication([])
150    loop = QEventLoop(app)
151    asyncio.set_event_loop(loop)
152    WPC_main_ui = MainWindow()
153    WPC_main_ui.show()
154    with loop:
155        loop.run_forever()
156
157if __name__ == "__main__":
158    main()
