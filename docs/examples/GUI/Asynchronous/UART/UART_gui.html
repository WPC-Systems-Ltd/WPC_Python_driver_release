  1##  Example_UART/ main.py
  2##  This is example for UART with WPC DAQ Device with asynchronous mode.

  6## Python
  7import asyncio
  8import os
  9from qasync import QEventLoop, asyncSlot
 10
 11## Third party
 12from PyQt5 import QtWidgets, QtGui
 13from UI_design.Ui_example_GUI_UART import Ui_MainWindow
 14
 15## WPC
 16from wpcsys import pywpc
 17
 18DEVIDER = 2000
 19class MainWindow(QtWidgets.QMainWindow):
 20    def __init__(self):
 21        super(MainWindow, self).__init__()
 22
 23        ## UI initialize
 24        self.ui = Ui_MainWindow()
 25        self.ui.setupUi(self)
 26
 27        ## Get Python driver version
 28        print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
 29
 30        ## Connection flag
 31        self.connect_flag = 0
 32
 33        ## Handle declaration
 34        self.dev = None
 35
 36        ## Material path
 37        file_path = os.path.dirname(__file__)
 38        self.trademark_path = file_path + "\Material\\trademark.jpg"
 39        self.blue_led_path = file_path + "\Material\LED_blue.png"
 40        self.red_led_path = file_path + "\Material\LED_red.png"
 41        self.green_led_path = file_path + "\Material\LED_green.png"
 42        self.gray_led_path = file_path + "\Material\LED_gray.png"
 43
 44        ## Set trademark & LED path
 45        self.ui.lb_trademark.setPixmap(QtGui.QPixmap(self.trademark_path))
 46        self.ui.lb_led.setPixmap(QtGui.QPixmap(self.gray_led_path))
 47        self.ui.lb_ledport.setPixmap(QtGui.QPixmap(self.gray_led_path))
 48
 49        ## Define callback events
 50        self.ui.btn_connect.clicked.connect(self.connectEvent)
 51        self.ui.btn_disconnect.clicked.connect(self.disconnectEvent)
 52
 53        self.ui.btn_open.clicked.connect(self.openPortEvent)
 54        self.ui.btn_close.clicked.connect(self.closePortEvent)
 55
 56        self.ui.btn_write.clicked.connect(self.writeEvent)
 57        self.ui.btn_read.clicked.connect(self.readEvent)
 58
 59    def selectHandle(self):
 60        handle_idx = int(self.ui.comboBox_handle.currentIndex())
 61        if handle_idx == 0:
 62            self.dev = pywpc.USBDAQF1D()
 63        elif handle_idx == 1:
 64            self.dev = pywpc.USBDAQF1AD()
 65        elif handle_idx == 2:
 66            self.dev = pywpc.USBDAQF1TD()
 67        elif handle_idx == 3:
 68            self.dev = pywpc.USBDAQF1RD()
 69        elif handle_idx == 4:
 70            self.dev = pywpc.USBDAQF1CD()
 71        elif handle_idx == 5:
 72            self.dev = pywpc.USBDAQF1AOD()
 73
 74    def updateParam(self):
 75        ## Get IP or serial_number from GUI
 76        self.ip = self.ui.lineEdit_IP.text()
 77
 78        ## Get port from GUI
 79        self.port = int(self.ui.comboBox_port.currentIndex())+1
 80
 81        # Get write data from GUI
 82        self.write_data = self.ui.lineEdit_write.text()
 83
 84        ## Get bytes to read from GUI
 85        self.read_bytes = self.ui.lineEdit_byteread.text()
 86
 87        ## Get baudrate from GUI
 88        self.baudrate = self.ui.comboBox_baudrate.currentText()
 89
 90        ## Get databit from GUI
 91        self.databit = self.ui.comboBox_databit.currentIndex()
 92
 93        ## Get parity from GUI
 94        self.parity = self.ui.comboBox_parity.currentIndex()
 95
 96        ## Get stop bit from GUI
 97        self.stopbit = self.ui.comboBox_stopbit.currentIndex()
 98
 99    @asyncSlot()
100    async def openPortEvent(self):
101        ## Update Param
102        self.updateParam()
103
104        ## Open UART port
105        status = await self.dev.UART_open_async(self.port)
106        print("UART_open_async status: ", status)
107
108        ## Set UART port and baudrate
109        status = await self.dev.UART_setBaudRate_async(self.port, int(self.baudrate))
110        print("UART_setBaudRate_async status: ", status)
111
112        ## Set UART port and data bit
113        status = await self.dev.UART_setDataBit_async(self.port, self.databit)
114        print("UART_setDataBit_async status: ", status)
115
116        ## Set UART port and parity
117        status = await self.dev.UART_setParity_async(self.port, self.parity)
118        print("UART_setParity_async status: ", status)
119
120        ## Set UART port and stop bit
121        status = await self.dev.UART_setNumStopBit_async(self.port, self.stopbit)
122        print("UART_setNumStopBit_async status: ", status)
123
124        ## Change port LED status
125        self.ui.lb_ledport.setPixmap(QtGui.QPixmap(self.blue_led_path))
126
127    @asyncSlot()
128    async def writeEvent(self):
129        ## Update Param
130        self.updateParam()
131
132        ## Set UART port and and write data to device
133        status = await self.dev.UART_write_async(self.port, self.write_data)
134        print("UART_write_async status: ", status)
135
136    @asyncSlot()
137    async def readEvent(self):
138        ## Update Param
139        self.updateParam()
140
141        ## Set UART port and read bytes
142        data = await self.dev.UART_read_async(self.port, int(self.read_bytes))
143        self.ui.lineEdit_read.setText(str(data))
144
145    @asyncSlot()
146    async def closePortEvent(self):
147        ## Update Param
148        self.updateParam()
149
150        ## Close UART port
151        status = await self.dev.UART_close_async(self.port)
152        print("UART_close_async status: ", status)
153
154        ## Change port LED status
155        self.ui.lb_ledport.setPixmap(QtGui.QPixmap(self.green_led_path))
156
157    @asyncSlot()
158    async def connectEvent(self):
159        if self.connect_flag == 1:
160            return
161
162        ## Select handle
163        self.selectHandle()
164
165        ## Update Param
166        self.updateParam()
167
168        ## Connect to device
169        try:
170            self.dev.connect(self.ip)
171        except pywpc.Error as err:
172            print("err: " + str(err))
173            return
174
175        ## Change LED status
176        self.ui.lb_led.setPixmap(QtGui.QPixmap(self.blue_led_path))
177
178        ## Change connection flag
179        self.connect_flag = 1
180
181    @asyncSlot()
182    async def disconnectEvent(self):
183        if self.connect_flag == 0:
184            return
185
186        ## Disconnect device
187        self.dev.disconnect()
188
189        ## Change LED status
190        self.ui.lb_led.setPixmap(QtGui.QPixmap(self.green_led_path))
191
192        ## Change connection flag
193        self.connect_flag = 0
194
195    def closeEvent(self, event):
196        if self.dev is not None:
197            ## Disconnect device
198            self.dev.disconnect()
199
200            ## Release device handle
201            self.dev.close()
202
203def main():
204    app = QtWidgets.QApplication([])
205    loop = QEventLoop(app)
206    asyncio.set_event_loop(loop)
207    WPC_main_ui = MainWindow()
208    WPC_main_ui.show()
209    with loop:
210        loop.run_forever()
211
212if __name__ == "__main__":
213    main()
