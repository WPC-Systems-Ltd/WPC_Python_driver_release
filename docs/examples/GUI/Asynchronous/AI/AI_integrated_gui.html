 79    def closeEvent(self, event):
 80        if self.dev is not None:
 81            ## Disconnect device
 82            self.disconnectEvent()
 83            ## Release device handle
 84            self.dev.close()
 85        else:
 86            self.killed = True
 87
 88    def selectHandle(self):
 89        handle_idx = int(self.ui.comboBox_handle.currentIndex())
 90        if handle_idx == 0:
 91            self.dev = pywpc.WifiDAQE3A()
 92        elif handle_idx == 1:
 93            self.dev = pywpc.EthanA()
 94        elif handle_idx == 2:
 95            self.dev = pywpc.USBDAQF1AD()
 96        elif handle_idx == 3:
 97            self.dev = pywpc.USBDAQF1AOD()
 98
 99    def updateParam(self):
100        ## Get IP or serial_number from GUI
101        self.ip = self.ui.lineEdit_IP.text()
102
103        ## Get port from GUI
104        self.port = int(self.ui.comboBox_port.currentIndex())
105
106        ## Get AI mode from UI
107        self.mode = self.ui.comboBox_aiMode.currentIndex()
108
109        ## Get Sampling rate from UI
110        self.ai_sampling_rate = float(self.ui.lineEdit_samplingRate.text())
111
112        ## Get NumSamples from UI
113        self.ai_n_samples = int(self.ui.lineEdit_numSamples.text())

115    @asyncSlot()
116    async def connectEvent(self):
117        if self.connect_flag == 1:
118            return
119
120        ## Select handle
121        self.selectHandle()
122
123        ## Update Param
124        self.updateParam()
125
126        try:
127            ## Connect to device
128            self.dev.connect(self.ip)
129        except pywpc.Error as err:
130            print("err: " + str(err))
131            return
132
133        ## Open AI port
134        status = await self.dev.AI_open_async(self.port)
135        print("AI_open_async status: ", status)
136
137        ## Change LED status
138        self.ui.lb_led.setPixmap(QtGui.QPixmap(self.blue_led_path))
139
140        ## loop start
141        self.killed = False
142        self.loop_fct()
143
144        ## Change connection flag
145        self.connect_flag = 1
146
147    @asyncSlot()
148    async def disconnectEvent(self):
149        if self.connect_flag == 0:
150            return
151
152        ## close AI port
153        status = await self.dev.AI_close_async(self.port)
154        print("AI_close_async status: ", status)
155
156        ## Disconnect device
157        self.dev.disconnect()
158
159        ## Change LED status
160        self.ui.lb_led.setPixmap(QtGui.QPixmap(self.green_led_path))
161
162        self.killed = True
163
164        ## Change connection flag
165        self.connect_flag = 0
166
167    @asyncSlot()
168    async def loop_fct(self):
169        while not self.killed:
170            ## data acquisition
171            data = await self.dev.AI_readStreaming_async(self.port, 600, 0.005)## Get 600 points at a time 
172            if len(data) >0 :
173                self.setDisplayPlotNums(data, self.ai_n_samples)
174            await asyncio.sleep(0.005)
175
176    @asyncSlot()
177    async def startAIEvent(self):
178        ## Check connection status
179        if self.checkConnectionStatus() == False:
180            return
181
182        ## Update Param
183        self.updateParam()
184
185        ## On Demand
186        if self.mode == 0:
187            data = await self.dev.AI_readOnDemand_async(self.port)
188            self.setDisplayPlotNums([data], self.ai_n_samples)
189        ## N-Samples/Continuous
190        else:
191            status = await self.dev.AI_start_async(self.port)
192            print("AI_start_async status: ", status)
193
194    @asyncSlot()

198            return

246    def plotInitial(self):
247        self.checkboxstatus = [1 for x in range(8)]
248        self.matplotlibwidget = MatplotlibWidget()
249        for i in range(8):
250            self.ui.MplWidget.canvas.axes.plot([0], [0], alpha=self.checkboxstatus[i])

252    def plotAnimation(self):
253        ## Rearrage x-axis data according to data amount
254        self.ani = animation.FuncAnimation(self.ui.MplWidget, self.plotChart, self.plotGetAxisData, interval=100, repeat=True)
255        self.ui.MplWidget.canvas.draw()
256
257    def plotGetAxisData(self):
258        # Get ch0~ch7 checkbox status from MainUI window
259        list_ch = [self.ui.cb_ch0, self.ui.cb_ch1, self.ui.cb_ch2, self.ui.cb_ch3, self.ui.cb_ch4, self.ui.cb_ch5,self.ui.cb_ch6, self.ui.cb_ch7]
260        for i in range(8):
261            self.checkboxstatus[i] = int(list_ch[i].isChecked())
262
263        ## Get xmin, xmax and x list
264        m = len(self.channel_list[0])
265        x_max = self.plot_total_times
266        x_min = max(x_max - m, 0)
267        x_list = list(range(x_min, x_max))
268
269        ## Get xticks
270        if m > 5:
271            dx = m // 6
272            ticks = np.arange(x_min, x_max, dx)
273        else:
274            ticks = np.arange(x_min, x_max)
275        yield x_list, ticks, x_min, x_max, self.checkboxstatus
276
277    def plotChart(self, update):
278        ## Define update value
279        x_list, ticks, x_min, x_max, self.checkboxstatus = update
280
281        ## Clear all axes info
282        self.ui.MplWidget.canvas.axes.clear()
283
284        ## Plot 8 channels data
285        try:
286            for i in range(8):
287                self.ui.MplWidget.canvas.axes.plot(x_list, self.channel_list[i], alpha=self.checkboxstatus[i],
288                                                marker='o', markersize=2)
289        except:
290            print("err_xlist " + str(len(x_list)))
291            print("err_ylist " + str(len(self.channel_list[i])))
292
293
294        ## Set x,y limit
295        self.ui.MplWidget.canvas.axes.set_ylim(float(self.plot_y_min) * 1.05, float(self.plot_y_max) * 1.05)
296        self.ui.MplWidget.canvas.axes.set_xlim(x_min, x_max)
297
298        ## Set xtickslabel
299        self.ui.MplWidget.canvas.axes.set_xticks(ticks)
300        new_ticks = self.plotConvertXtick(ticks)
301        self.ui.MplWidget.canvas.axes.set_xticklabels(new_ticks)
302
303        ## Set label
304        self.ui.MplWidget.canvas.axes.set_xlabel("Time (s)", fontsize=12)
305        self.ui.MplWidget.canvas.axes.set_ylabel("Voltage (V)", fontsize=12)
306
307        ## Set legend
308        self.ui.MplWidget.canvas.axes.legend(('ch0', 'ch1', 'ch2', 'ch3', 'ch4', 'ch5', 'ch6', 'ch7'),
309                                            loc='center left', shadow=False, fontsize=10, bbox_to_anchor=(1, 0.75),
310                                            facecolor='#f0f0f0')
311        ## Set grid
312        self.ui.MplWidget.canvas.axes.grid(color='#bac3d1', linestyle='-', linewidth=0.8)  # grid
313
314    def plotConvertXtick(self, xtick):
315        return ["{:.2f}".format(x / self.ai_sampling_rate) for i, x in enumerate(xtick)]
316
317    def setDisplayPlotNums(self, data, nums):
318        data = np.array(data)
319        self.plot_total_times += len(data)
320        ## for plotting
321        for k in range(8):
322            self.channel_list[k].extend(data[:, k])
323            self.channel_list[k] = self.channel_list[k][-(nums):]
