 7async def main():
 8    print("Start example code...")

27        
28        ## Get pinmode from port 0 to port 3
29        for i in range(4):
30            ## Get pin mode
31            pin_mode = await dev.Sys_getPinModeInPort(i)
32            print("pin_mode", pin_mode)
33
34            ## Wait for 0.5 seconds
35            await asyncio.sleep(0.5)  ## delay(second)
36        print()
37        print("====================")
38
39        ## Parameters setting
40        port_DO = 0
41        port_DI = 1
42        
43        ## Open pin0, pin1, pin2, pin3 and pin4 in port 0 with digital output
44        await dev.DO_openPins(port_DO, [0,1,2,3,4])
45
46        ## Set pin0, pin3 and pin4 to digital high, others to digital low
47        await dev.DO_writeValuePins(port_DO, [0,1,2,3,4], [1,0,0,1,1]) 
48
49        ## Open pin4, pin5, pin6 and pin7 in port 1 with digital input

59            await asyncio.sleep(0.5)  ## delay(second)
60
61        ## Wait for 1 seconds

64        ## Set port 0 to idle
