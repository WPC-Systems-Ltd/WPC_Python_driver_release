 1import asyncio
 2import sys
 3sys.path.insert(0, 'pywpc/')
 4sys.path.insert(0, '../../../pywpc/')
 5import pywpc
 6
 7async def main():
 8    print("Start example code...")
 9
10    ## Get Python driver version
11    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
12
13    ## Create device handle
14    dev = pywpc.USBDAQF1D()
15
16    ## Connect to network device
17    try:
18        dev.connect('21JA1044')
19    except Exception as err:
20        pywpc.printGenericError(err)
21
22    try: 
23        ## Get firmware model & version
24        driver_info = await dev.Sys_getDriverInfo()
25        print("Model name: " + driver_info[0])
26        print("Firmware version: " + driver_info[-1])
27
28        ## Parameters setting
29        port = 2
30        baudrate = 9600 
31        data_bit_mode = 0  ## 0 : 8-bit data, 1 : 9-bit data.
32        parity_mode = 0    ## 0 : None, 2 : Even parity, 3 : Odd parity.
33        stop_bit_mode = 0  ## 0 : 1 bit, 1 : 0.5 bits, 2 : 2 bits, 3 : 1.5 bits  
34       
35        ## Open port 2
36        status = await dev.UART_open(port) 
37        if status == 0: print("UART_open: OK")
38
39        ## Set baudrate to 9600
40        status = await dev.UART_setBaudRate(port, baudrate)
41        if status == 0: print("UART_setBaudRate: OK")
42 
43        ## Set data bit to 8-bit data
44        status = await dev.UART_setDataBit(port, data_bit_mode)
45        if status == 0: print("UART_setDataBit: OK")
46        
47        ## Set parity to None
48        status = await dev.UART_setParity(port, parity_mode)
49        if status == 0: print("UART_setParity: OK")
50
51        ## Set stop bit to 8-bit data
52        status = await dev.UART_setNumStopBit(port, stop_bit_mode)
53        if status == 0: print("UART_setNumStopBit: OK")
54
55        ## Sleep
56        await asyncio.sleep(10) ## delay(second)
57        
58        ## Read 20 bytes
59        data = await dev.UART_read(port, 20) 
60        print("data: ", data)
61
62        ## Close port 2
63        status = await dev.UART_close(port) 
64        if status == 0: print("UART_close: OK")
65
66    except Exception as err:
67        pywpc.printGenericError(err)
68
69    ## Disconnect network device
70    dev.disconnect()
71
72    ## Release device handle
73    dev.close()
74    
75    print("End example code...")
76    return
77
78if __name__ == '__main__':
79    asyncio.run(main())
