 1import asyncio
 2import sys
 3sys.path.insert(0, 'pywpc/')
 4import pywpc
 5
 6async def main():
 7    print("Start example code...")
 8
 9    ## Get Python driver version
10    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
11
12    ## Create device handle
13    dev = pywpc.USBDAQF1D()
14
15    ## Connect to network device
16    try:
17        dev.connect('21JA1044')
18    except Exception as err:
19        pywpc.printGenericError(err)
20
21    try: 
22        ## Get firmware model & version
23        driver_info = await dev.Sys_getDriverInfo()
24        print("Model name: " + driver_info[0])
25        print("Firmware version: " + driver_info[-1])
26        
27        ## Parameters setting
28        port = 2
29        baudrate = 9600 
30        data_bit_mode = 0  ## 0 : 8-bit data, 1 : 9-bit data.
31        parity_mode = 0    ## 0 : None, 2 : Even parity, 3 : Odd parity.
32        stop_bit_mode = 0  ## 0 : 1 bit, 1 : 0.5 bits, 2 : 2 bits, 3 : 1.5 bits  
33
34        ## Open UART port2
35        status = await dev.UART_open(port) 
36        if status == 0: print("UART_open: OK")
37
38        ## Set UART port to 2 and set baudrate to 9600
39        status = await dev.UART_setBaudRate(port, baudrate)
40        if status == 0: print("UART_setBaudRate: OK")
41 
42        ## Set UART port to 2 and set data bit to 8-bit data
43        status = await dev.UART_setDataBit(port, data_bit_mode)
44        if status == 0: print("UART_setDataBit: OK")
45        
46        ## Set UART port to 2 and set parity to None
47        status = await dev.UART_setParity(port, parity_mode)
48        if status == 0: print("UART_setParity: OK")
49
50        ## Set UART port to 2 and set stop bit to 8-bit data
51        status = await dev.UART_setNumStopBit(port, stop_bit_mode)
52        if status == 0: print("UART_setNumStopBit: OK")
53
54        ## Set UART port to 2 and and write "12345" to device in string format
55        status = await dev.UART_write(port, "12345")
56        if status == 0: print("UART_write: OK")
57
58        ## Set UART port to 2 and and write "chunglee people" to device
59        status = await dev.UART_write(port, "chunglee people")
60        if status == 0: print("UART_write: OK")
61
62        ## Set UART port to 2 and and write "12345" to device in list format
63        status = await dev.UART_write(port, ["1","2","3","4","5"])
64        if status == 0: print("UART_write: OK")
65        
66        ## Close UART port2
67        status = await dev.UART_close(port)
68        if status == 0: print("UART_close: OK")
69    except Exception as err:
70        pywpc.printGenericError(err)
71
72    ## Disconnect network device
73    dev.disconnect()
74
75    ## Release device handle
76    dev.close()
77    
78    print("End example code...")
79    return
80
81if __name__ == '__main__':
82    asyncio.run(main())
