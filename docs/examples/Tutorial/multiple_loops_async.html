 6## Python
 7import asyncio
 8import sys
 9
10## WPC
11sys.path.insert(0, 'pywpc/')
12sys.path.insert(0, '../../../pywpc/')
13import pywpc
14
15async def readRSSI_loop(handle, delay = 1):
16    while True:
17        data = await handle.Wifi_readRSSI()
18        print("RSSI: " + str(data) + " dBm")
19        await asyncio.sleep(delay)  ## delay(second)
20
21async def readBattery_loop(handle, delay= 1):
22    while True:
23        data = await handle.Wifi_readBattery()
24        print("Battery: "+ str(data) + " mV")
25        await asyncio.sleep(delay)  ## delay(second)
26
27async def main(): 
28    print("Start example code...")
29
30    ## Get Python driver version
31    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}') 
32
33    ## Create device handle
34    dev = pywpc.WifiDAQE3A()
35
36    ## Connect to network device
37    try:
38        dev.connect("192.168.5.79")
39    except Exception as err:
40        pywpc.printGenericError(err)
41
42    ## Perform two async thread to get data
43    try:
44        ## Get firmware model & version
45        driver_info = await dev.Sys_getDriverInfo()
46        print("Model name: " + driver_info[0])
47        print("Firmware version: " + driver_info[-1])
48        
49        await asyncio.gather(readRSSI_loop(dev, 1), readBattery_loop(dev, 2)) ## delay (second)
50    except Exception as err:
51        pywpc.printGenericError(err)
52
53    ## This part never execute because the async thread.
54   
55    ## Disconnect network device
56    dev.disconnect()
57    
58    ## Release device handle
59    dev.close()
60
61    print("End example code...")
62    return
63
64if __name__ == '__main__': 
65    asyncio.run(main())
