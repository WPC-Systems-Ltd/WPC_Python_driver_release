example I2C write and readÔÉÅ

  1'''
  2I2C - example_I2C_write_read.py
  3
  4This example demonstrates how to communicate with USB-DAQ-F1-D (master) and 24C08C (slave) with I2C interface.
  5
  6First, it shows how to open I2C port and configure I2C parameters.
  7Second, write some bytes with address into EEPROM (24C08C). We have to make sure that bytes written in address is correct however read address from EEPROM (24C08C).
  8Last, close I2C port
  9
 10The sensor used in this example is a 24C08C expecially for Two-wore Serial EEPROM
 11
 12For other examples please check:
 13   https://github.com/WPC-Systems-Ltd/WPC_Python_driver_release/tree/main/Examples
 14
 15   See README.md file to get detailed usage of this example.
 16
 17Copyright (c) 2022 WPC Systems Ltd.
 18All rights reserved.
 19
 20'''
 21
 22## Python
 23import asyncio
 24
 25## WPC
 26from wpcsys import pywpc
 27
 28async def main():
 29    print("Start example code...")
 30
 31    ## Get Python driver version
 32    print(f'{pywpc.PKG_FULL_NAME} - Version {pywpc.__version__}')
 33
 34    ## Create device handle
 35    dev = pywpc.USBDAQF1D()
 36
 37    ## Connect to USB device
 38    try:
 39        dev.connect("21JA1279")
 40    except Exception as err:
 41        pywpc.printGenericError(err)
 42        
 43    try: 
 44        ## Get firmware model & version
 45        driver_info = await dev.Sys_getDriverInfo_async()
 46        print("Model name: " + driver_info[0])
 47        print("Firmware version: " + driver_info[-1])
 48
 49        '''
 50        Take 24C08N for example
 51        '''
 52
 53        ## Parameters setting
 54        I2C_port = 1
 55        mode = 0
 56        device_address = 0x50 ## 01010000
 57        word_address = 0x00
 58
 59        '''
 60        Open I2C port
 61        ''' 
 62
 63        ## Open I2C port1
 64        status = await dev.I2C_open_async(I2C_port)
 65        if status == 0: print("I2C_open: OK")
 66        
 67        '''
 68        Set I2C parameter
 69        '''
 70
 71        ## Set I2C port to 1 and set clock rate to standard mode
 72        status = await dev.I2C_setClockRate_async(I2C_port, mode)
 73        if status == 0: print("I2C_setClockRate: OK")
 74
 75        '''
 76        Write data via I2C
 77        '''
 78        
 79        ## Write WREN byte
 80        status = await dev.I2C_write_async(I2C_port, device_address, [word_address, 0xAA, 0x55, 0xAA, 0x55])
 81        if status == 0: print("I2C_write: OK")
 82        
 83        ## Sleep
 84        await asyncio.sleep(0.1) ## delay(second)
 85
 86        '''
 87        Read data via I2C
 88        '''
 89
 90        status = await dev.I2C_write_async(I2C_port, device_address, [word_address])
 91        if status == 0: print("I2C_write: OK")
 92
 93        ## Sleep
 94        await asyncio.sleep(0.1) ## delay(second)
 95
 96        data_list = await dev.I2C_read_async(I2C_port, device_address, 4)
 97        print("read data :", data_list)
 98       
 99        '''
100        Close I2C port
101        ''' 
102
103        ## Close I2C port1
104        status = await dev.I2C_close_async(I2C_port)
105        if status == 0: print("I2C_close: OK")
106    except Exception as err:
107        pywpc.printGenericError(err)
108
109    ## Disconnect USB device
110    dev.disconnect()
111
112    ## Release device handle
113    dev.close()
114
115    print("End example code...")
116    return
117    
118if __name__ == '__main__':
119    asyncio.run(main())
